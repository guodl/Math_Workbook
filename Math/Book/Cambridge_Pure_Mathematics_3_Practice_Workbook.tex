\begin{bxExample}{Differentiation - Product Rule}    
    \textbf{Example} \indent
	\begin{itemize}
	     \item[] Differentiate $y=e^x \sin x.
	\end{itemize}
\tcbline 

Solution:\indent \indent \indent \indent
\vspace{0.5cm} 

$\begin{WithArrows}%[format = l]
       \qquad           y'    &=  (e^x)' \sin x + e^x (\sin x)' 	      \Arrow{Product Rule $(uv)'=u'v+uv'$}\\[10pt]
	                          &=  e^x \sin x + e^x \cos x                 \Arrow{Factorising} \\[10pt]
                              &=  e^x(\sin x + \cos x)                                             
\end{WithArrows}$

\tcbline

Note:\\
\begin{warningEnv}
    $\begin{WithArrows}%[format = l]
        \qquad y  &=  e^x sinx          \Arrow{ $ $}\\[10pt]
		       y' &=  (e^x)' (sinx)'   \\
	              &=  e^x cosx
    \end{WithArrows}$
	
\end{warningEnv}

\end{bxExample}
