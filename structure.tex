\newcounter{theo}[section] \setcounter{theo}{0}
\renewcommand{\thetheo}{\arabic{section}.\arabic{theo}}
\newenvironment{theo}[2][]{%
\refstepcounter{theo}%
\ifstrempty{#1}%
{\mdfsetup{%
frametitle={%
\tikz[baseline=(current bounding box.east),outer sep=0pt]
\node[anchor=east,rectangle,fill=blue!20]
{\strut Theorem~\thetheo};}}
}%
{\mdfsetup{%
frametitle={%
\tikz[baseline=(current bounding box.east),outer sep=0pt]
\node[anchor=east,rectangle,fill=blue!20]
{\strut Theorem~\thetheo:~#1};}}%
}%
\mdfsetup{innertopmargin=10pt,linecolor=blue!20,%
linewidth=2pt,topline=true,%
frametitleaboveskip=\dimexpr-\ht\strutbox\relax
}
\begin{mdframed}[]\relax%
\label{#2}}{\end{mdframed}}

    frametitle={%

    \node[anchor=east,rectangle,fill=green!20]

    frametitle={%
    \tikz[baseline=(current bounding box.east),outer sep=0pt]
    \node[anchor=east,rectangle,fill=green!20]
    {\strut Lemma~\thelem:~#1};}}%
}%
\mdfsetup{innertopmargin=10pt,linecolor=green!20,%
    linewidth=2pt,topline=true,%
    frametitleaboveskip=\dimexpr-\ht\strutbox\relax
}
\begin{mdframed}[]\relax%
    \label{#2}}{\end{mdframed}}

\NewEnviron{envAnswer}[1][]
{
    \pgfkeys{/envAnswer,default, #1}
    
    \ifnum \blankline>0
        \ifboolexpr{not ( togl{bSolution} or togl{bCompress}) } { 
            \Pointilles[1.5]{\blankline}
			\vspace{0.3cm}
        }
    
    \fi
    
    \iftoggle{bAnswer}{ 
	    %-- To display answer
        \hfill \answer \udensdot[\length]{
                                  \colorbox{green!30!white}{\BODY}
                                }
    }{  %-- Set to white color to hide answer
        \hfill \answer \udensdot[\length]{
                                  \textcolor{white}{\BODY}
                                }
	}
